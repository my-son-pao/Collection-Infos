<script>

    let allRecords = [];


    // ========== 加载数据 ==========

    async function loadData() {

        const listEl = document.getElementById('recordList');

        listEl.innerHTML = `

            <div class="loading-state">

                <div class="loading-spinner"></div>

                <p>加载中...</p>

            </div>

        `;


        try {

            const response = await fetch(

                `${SUPABASE_URL}/rest/v1/submissions?select=*&order=created_at.desc`,

                {

                    headers: {

                        'Authorization': `Bearer ${SUPABASE_KEY}`,

                        'apikey': SUPABASE_KEY

                    }

                }

            );


            if (!response.ok) throw new Error('加载失败');


            allRecords = await response.json();

            updateStats();

            renderRecords(allRecords);


        } catch (error) {

            listEl.innerHTML = `

                <div class="empty-state">

                    <div class="icon">❌</div>

                    <p>加载失败，请检查配置</p>

                    <p style="font-size:12px;margin-top:8px;color:#e74c3c">${error.message}</p>

                </div>

            `;

        }

    }


    // ========== 更新统计 ==========

    function updateStats() {

        const total = allRecords.length;

        const today = new Date().toDateString();

        const todayCount = allRecords.filter(r =>

            new Date(r.created_at).toDateString() === today

        ).length;

        const photosCount = allRecords.reduce((sum, r) =>

            sum + (r.photos ? r.photos.length : 0), 0

        );


        document.getElementById('totalBadge').textContent = total + ' 条';

        document.getElementById('statTotal').textContent = total;

        document.getElementById('statToday').textContent = todayCount;

        document.getElementById('statPhotos').textContent = photosCount;

    }


    // ========== 渲染记录 ==========

    function renderRecords(records) {

        const listEl = document.getElementById('recordList');


        if (records.length === 0) {

            listEl.innerHTML = `

                <div class="empty-state">

                    <div class="icon">📭</div>

                    <p>暂无提交记录</p>

                </div>

            `;

            return;

        }


        listEl.innerHTML = records.map((r, index) => {

            const time = new Date(r.created_at).toLocaleString('zh-CN');

            const photos = r.photos || [];

            const hasRemark = r.remark && r.remark.trim();


            return `

                <div class="record-card">

                    <div class="record-header">

                        <div>

                            <span class="person-name">${escapeHtml(r.name)}</span>

                            <span class="record-id">#${r.id}</span>

                        </div>

                        <div>

                            <span class="time">${time}</span>

                            <button class="delete-btn" onclick="deleteRecord(${r.id}, this)">删除</button>

                        </div>

                    </div>


                    <div class="info-grid">

                        <div class="info-item">

                            <div class="info-label">手机号</div>

                            <div class="info-value">${escapeHtml(r.phone)}</div>

                        </div>

                        <div class="info-item">

                            <div class="info-label">身份证号</div>

                            <div class="info-value">${r.idcard ? escapeHtml(r.idcard) : '—'}</div>

                        </div>

                    </div>


                    ${hasRemark ? `

                        <div class="remark-box" style="display:block">

                            💬 ${escapeHtml(r.remark)}

                        </div>

                    ` : ''}


                    <div class="photo-grid">

                        ${photos.map(url => `

                            <img src="${url}" onclick="openLightbox('${url}')" loading="lazy">

                        `).join('')}

                    </div>

                </div>

            `;

        }).join('');

    }


    // ========== 搜索过滤 ==========

    function filterRecords() {

        const keyword = document.getElementById('searchInput').value.trim().toLowerCase();

        if (!keyword) {

            renderRecords(allRecords);

            return;

        }

        const filtered = allRecords.filter(r =>

            r.name.toLowerCase().includes(keyword) ||

            r.phone.includes(keyword) ||

            (r.idcard && r.idcard.includes(keyword))

        );

        renderRecords(filtered);

    }


    // ========== 删除记录 ==========

    async function deleteRecord(id, btnEl) {

        if (!confirm('确定删除这条记录吗？删除后不可恢复！')) return;


        btnEl.textContent = '删除中...';

        btnEl.disabled = true;


        try {

            const response = await fetch(

                `${SUPABASE_URL}/rest/v1/submissions?id=eq.${id}`,

                {

                    method: 'DELETE',

                    headers: {

                        'Authorization': `Bearer ${SUPABASE_KEY}`,

                        'apikey': SUPABASE_KEY

                    }

                }

            );


            if (!response.ok) throw new Error('删除失败');


            // 重新加载

            loadData();

        } catch (error) {

            alert('删除失败: ' + error.message);

            btnEl.textContent = '删除';

            btnEl.disabled = false;

        }

    }


    // ========== 导出 CSV ==========

    function exportCSV() {

        if (allRecords.length === 0) {

            alert('暂无数据可导出');

            return;

        }


        let csv = '\uFEFF姓名,手机号,身份证号,备注,照片链接,提交时间\n';

        allRecords.forEach(r => {

            const photos = (r.photos || []).join(' | ');

            const time = new Date(r.created_at).toLocaleString('zh-CN');

            csv += `"${r.name}","${r.phone}","${r.idcard || ''}","${r.remark || ''}","${photos}","${time}"\n`;

        });


        const blob = new Blob([csv], { type: 'text/csv;charset=utf-8' });

        const a = document.createElement('a');

        a.href = URL.createObjectURL(blob);

        a.download = `信息采集数据_${new Date().toLocaleDateString('zh-CN')}.csv`;

        a.click();

        URL.revokeObjectURL(a.href);

    }


    // ========== 图片预览 ==========

    function openLightbox(url) {

        document.getElementById('lightboxImg').src = url;

        document.getElementById('lightbox').classList.add('active');

    }


    function closeLightbox() {

        document.getElementById('lightbox').classList.remove('active');

    }


    // ESC 关闭预览

    document.addEventListener('keydown', (e) => {

        if (e.key === 'Escape') closeLightbox();

    });


    // ========== 防 XSS ==========

    function escapeHtml(str) {

        const div = document.createElement('div');

        div.textContent = str;

        return div.innerHTML;

    }


    // ========== 页面加载 ==========

    loadData();


    // 每30秒自动刷新

    setInterval(loadData, 30000);
