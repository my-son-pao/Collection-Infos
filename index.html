<script>

    // ========== 全局变量 ==========

    let selectedFiles = []; // 存储选中的文件对象

    const MAX_PHOTOS = 5;


    // ========== Toast 提示 ==========

    function showToast(msg, isError = false) {

        const toast = document.getElementById('toast');

        toast.textContent = msg;

        toast.className = isError ? 'toast error show' : 'toast show';

        setTimeout(() => { toast.className = 'toast'; }, 3000);

    }


    // ========== 图片压缩 ==========

    function compressImage(file, maxWidth = 1200, quality = 0.75) {

        return new Promise((resolve) => {

            const reader = new FileReader();

            reader.onload = function(e) {

                const img = new Image();

                img.onload = function() {

                    const canvas = document.createElement('canvas');

                    let w = img.width, h = img.height;

                    if (w > maxWidth) {

                        h = (maxWidth / w) * h;

                        w = maxWidth;

                    }

                    canvas.width = w;

                    canvas.height = h;

                    canvas.getContext('2d').drawImage(img, 0, 0, w, h);

                    canvas.toBlob((blob) => resolve(blob), 'image/jpeg', quality);

                };

                img.src = e.target.result;

            };

            reader.readAsDataURL(file);

        });

    }


    // ========== 照片选择 ==========

    document.getElementById('photoInput').addEventListener('change', function(e) {

        const files = Array.from(e.target.files);


        if (selectedFiles.length + files.length > MAX_PHOTOS) {

            showToast(`最多只能上传${MAX_PHOTOS}张照片`, true);

            return;

        }


        files.forEach(file => {

            if (!file.type.startsWith('image/')) return;

            const preview = URL.createObjectURL(file);

            selectedFiles.push({ file, preview });

        });


        renderPreviews();

        e.target.value = ''; // 清空 input，允许再次选择相同文件

    });


    function renderPreviews() {

        const grid = document.getElementById('previewGrid');

        grid.innerHTML = '';

        selectedFiles.forEach((item, index) => {

            const div = document.createElement('div');

            div.className = 'preview-item';

            div.innerHTML = `

                <img src="${item.preview}">

                <button type="button" class="remove-btn" onclick="removePhoto(${index})">✕</button>

            `;

            grid.appendChild(div);

        });

    }


    function removePhoto(index) {

        URL.revokeObjectURL(selectedFiles[index].preview);

        selectedFiles.splice(index, 1);

        renderPreviews();

    }


    // ========== 上传图片到 Supabase Storage ==========

    async function uploadPhoto(file) {

        const compressed = await compressImage(file);

        const fileName = `${Date.now()}_${Math.random().toString(36).substr(2, 8)}.jpg`;


        const response = await fetch(

            `${SUPABASE_URL}/storage/v1/object/photos/${fileName}`,

            {

                method: 'POST',

                headers: {

                    'Authorization': `Bearer ${SUPABASE_KEY}`,

                    'apikey': SUPABASE_KEY,

                    'Content-Type': 'image/jpeg',

                    'x-upsert': 'true'

                },

                body: compressed

            }

        );


        if (!response.ok) {

            const err = await response.text();

            throw new Error('图片上传失败: ' + err);

        }


        // 返回公开访问URL

        return `${SUPABASE_URL}/storage/v1/object/public/photos/${fileName}`;

    }


    // ========== 保存记录到 Supabase 数据库 ==========

    async function saveRecord(data) {

        const response = await fetch(

            `${SUPABASE_URL}/rest/v1/submissions`,

            {

                method: 'POST',

                headers: {

                    'Authorization': `Bearer ${SUPABASE_KEY}`,

                    'apikey': SUPABASE_KEY,

                    'Content-Type': 'application/json',

                    'Prefer': 'return=minimal'

                },

                body: JSON.stringify(data)

            }

        );


        if (!response.ok) {

            const err = await response.text();

            throw new Error('数据保存失败: ' + err);

        }

    }


    // ========== 表单提交 ==========

    document.getElementById('collectForm').addEventListener('submit', async function(e) {

        e.preventDefault();


        const name = document.getElementById('name').value.trim();

        const phone = document.getElementById('phone').value.trim();

        const idcard = document.getElementById('idcard').value.trim();

        const remark = document.getElementById('remark').value.trim();


        // 验证

        if (!name) { showToast('请输入姓名', true); return; }

        if (!phone || !/^1\d{10}$/.test(phone)) { showToast('请输入正确的手机号', true); return; }

        if (selectedFiles.length === 0) { showToast('请至少上传一张照片', true); return; }


        const btn = document.getElementById('submitBtn');

        btn.disabled = true;

        btn.innerHTML = '<span class="spinner"></span>提交中...';


        try {

            // 1. 依次上传所有照片

            const photoUrls = [];

            for (let i = 0; i < selectedFiles.length; i++) {

                btn.innerHTML = `<span class="spinner"></span>上传照片 ${i + 1}/${selectedFiles.length}`;

                const url = await uploadPhoto(selectedFiles[i].file);

                photoUrls.push(url);

            }


            // 2. 保存表单数据

            btn.innerHTML = '<span class="spinner"></span>保存信息...';

            await saveRecord({

                name,

                phone,

                idcard,

                remark,

                photos: photoUrls

            });


            // 3. 显示成功

            document.getElementById('formCard').style.display = 'none';

            document.getElementById('successPage').style.display = 'block';


        } catch (error) {

            console.error(error);

            showToast('提交失败: ' + error.message, true);

            btn.disabled = false;

            btn.textContent = '提 交';

        }

    });


    // ========== 重置表单 ==========

    function resetForm() {

        document.getElementById('collectForm').reset();

        selectedFiles = [];

        renderPreviews();

        document.getElementById('formCard').style.display = 'block';

        document.getElementById('successPage').style.display = 'none';

        document.getElementById('submitBtn').disabled = false;

        document.getElementById('submitBtn').textContent = '提 交';

    }

</script>
